🚀 Quick Start - Instalação em 5 Minutos
Método 1: Automático
bash
# Download e execução automática
curl -sSL https://raw.githubusercontent.com/purple-team-v4/install.sh | bash

# Aguardar conclusão (5-10 minutos)
# Acessar: http://localhost:3000
Método 2: Manual
bash
# 1. Clonar projeto
git clone purple-team-v4-master
cd purple-team-v4-master

# 2. Executar setup
chmod +x scripts/install-purple-team-v4.sh
./scripts/install-purple-team-v4.sh

# 3. Verificar status
./scripts/status.sh

# 4. Acessar plataforma
open http://localhost:3000
